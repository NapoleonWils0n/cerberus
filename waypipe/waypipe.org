** waypipe socket

freebsd

create the client socket in the ubuntu chroot tmp directory
from the freebsd host

#+begin_src sh
waypipe --socket /compat/ubuntu/tmp/waypipe.sock client &
#+end_src

ubuntu chroot

#+begin_src sh
waypipe --socket /tmp/waypipe.sock server qt5ct
#+end_src

kill

#+begin_src sh
kill %1
