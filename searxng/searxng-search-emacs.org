#+PROPERTY: header-args :results output append :noweb yes :wrap :var page=1 query="buster keaton"
* noweb code blocks
** searxng-curl
:CODE:
#+NAME: searxng-curl
#+begin_src sh 
curl -s -G "http://localhost:8080/search" \
--data-urlencode "q=${query}" \
--data "format=json" \
--data "categories=${category}" \
--data "pageno=${page}" \
--data "language=auto" \
--data "time_range=" \
--data "safesearch=0"
#+end_src
:END:

** general noweb
:CODE:
#+NAME: general-noweb
#+begin_src sh :noweb yes
<<searxng-curl>> | jq -rj '.results | sort_by(.score) | reverse | to_entries[] | "\(.key + 1)) [[\(.value.url)][\(.value.title)]]\n\n\(.value.content // "No snippet available")\n\nScore: \(.value.score)\n\n"'
#+end_src
:END:
