fetch -o ./external/spirv/unified1/spirv.hpp https://raw.githubusercontent.com/KhronosGroup/SPIRV-Headers/main/include/spirv/unified1/spirv.hpp

remove the build directory if it exists

#+begin_src sh
rm -rf build
#+end_src

build whisper-cpp with vulkan

Configure CMake to build static libraries

#+begin_src sh
cmake -B build \
  -DGGML_VULKAN=ON \
  -DBUILD_SHARED_LIBS=OFF \
  -DWHISPER_ALL_EXTRAS=ON \
  -DCMAKE_CXX_FLAGS="-I$(pwd)/external"
#+end_src

Build the binary (adjust -j for your CPU cores)

#+begin_src sh
cmake --build build -j$(sysctl -n hw.ncpu) --config Release
#+end_src

install to whisper-cli ~/bin

#+begin_src sh
cp build/bin/whisper-cli ~/bin/
#+end_src

** Audio Format

whisper.cpp is extremely fast, but it is "picky" about its input.

It requires 16-bit WAV files at a 16kHz sample rate.

#+begin_src sh
ffmpeg -i video.mp4 -vn -ar 16000 -ac 1 -c:a pcm_s16le audio.wav
#+end_src

#+begin_src sh
whisper-cli -m ~/.local/share/whisper-models/ggml-large-v3-q4_0.bin -f audio.wav -otxt -osrt -ovtt 
#+end_src

** transribe-audio script

transcribe-audio script

#+begin_src sh
#!/bin/sh

#===============================================================================
# Wrapper for whisper-cpp to handle local library path
#===============================================================================

#===============================================================================
# script usage
#===============================================================================

usage () {
[ -z "${1}" ] || echo "! ${1}"
echo "Usage: $(basename "$0") -m model -f input -otxt -osrt -ovtt

if you omit the -m option the default model will be used
${HOME}/.local/share/whisper-models/ggml-small.en.bin"
exit 1
}


#===============================================================================
# error messages
#===============================================================================

WRONG_ARGS_ERR='wrong number of arguments passed to script'
INVALID_OPT_ERR='Invalid option:'
REQ_ARG_ERR='requires an argument'
NOTFILE_ERR='not a file'


#===============================================================================
# check number of aruments passed to script
#===============================================================================

[ $# -gt 0 ] || usage "${WRONG_ARGS_ERR}"


#===============================================================================
# getopts check options passed to script
#===============================================================================

while getopts ':m:f:h' opt
do
  case ${opt} in
     m) model="${OPTARG}"
        [ -f "${model}" ] || usage "${model} ${NOTFILE_ERR}";;
     f) input="${OPTARG}"
        [ -f "${input}" ] || usage "${input} ${NOTFILE_ERR}";;
     h) usage;;
     \?) OPTIND=$((OPTIND - 1)) # Handle passthrough of whisper-cpp options
        break;;
     :) usage "${INVALID_OPT_ERR} ${OPTARG} ${REQ_ARG_ERR}" 1>&2;;
  esac
done
shift $((OPTIND-1))


#===============================================================================
# input name
#===============================================================================

input_nopath="${input##*/}"
input_name="${input_nopath%.*}"


#===============================================================================
# check if default model exists
#===============================================================================

model_default="${HOME}/.local/share/whisper-models/ggml-small.en.bin"
[ -f "${model_default}" ] || usage "${model_default} ${NOTFILE_ERR}"


#===============================================================================
# function
#===============================================================================

transcribe () {
exec whisper-cli -np -m "${model:=${model_default}}" -f "${input}" -of "${input_name}" "${@}"
}


#===============================================================================
# run function
#===============================================================================

transcribe "${@}"
#+end_src
