** mount zfs snapshot

create a mount point 

#+begin_src sh
mkdir -p ~/mnt
#+end_src

mount a zfs snapshot on the mount point

#+begin_src sh
doas mount -t zfs zroot/home/djwilcox/desktop@29-04-2026 /home/djwilcox/mnt
#+end_src

copy files from the mount point to the desktop

#+begin_src sh
cp ~/mnt/great-ideas.org ~/desktop
#+end_src

unmount the mount point

#+begin_src sh
umount ~/mnt
#+end_src
