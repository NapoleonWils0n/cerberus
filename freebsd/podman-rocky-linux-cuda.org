devfsrules_jail=4 is the rule for podman containers

#+begin_src conf
[localrules=5]
add path 'da*' mode 0660 group operator
add path 'dri/*' mode 0660 group video
add path 'drm/*' mode 0660 group video
add path 'input/*' mode 0660 group video
add path 'input/event*' mode 0660 group video
add path 'nvidia*' mode 0660 group video

[devfsrules_jail=4]
add include $devfsrules_unhide_basic
add include $devfsrules_unhide_login
add path 'mixer*' unhide
add path 'dsp*' unhide
add path 'dri*' unhide
add path 'drm*' unhide
add path 'nvidia*' unhide
add path 'speaker*' unhide

#include <dlfcn.h>
#include <fcntl.h>

#include <unistd.h>

#define NV_UVM_INITIALIZE   0x30000001
#define NV_UVM_DEINITIALIZE 0x30000002
#define NV_ERR_NOT_SUPPORTED 0x56

struct NvUvmInitParams
{
  uint64_t flags __attribute__((aligned(8)));
  uint32_t status;
};

// ioctl interception - unchanged from shkhln's original

int (*libc_ioctl)(int fd, unsigned long request, ...) = NULL;

int ioctl(int fd, unsigned long request, ...) {
  if (!libc_ioctl) libc_ioctl = dlsym(RTLD_NEXT, "ioctl");

  va_list _args_;
  va_start(_args_, request);
  void* data = va_arg(_args_, void*);
  va_end(_args_);

  if (request == NV_UVM_INITIALIZE) {
    struct NvUvmInitParams* params = (struct NvUvmInitParams*)data;
    params->status = NV_ERR_NOT_SUPPORTED;
    return 0;
  }
  if (request == NV_UVM_DEINITIALIZE) return 0;

  return libc_ioctl(fd, request, data);
}

// path checks

static int is_nvidia_uvm(const char* path) {
  return path && strcmp("/dev/nvidia-uvm", path) == 0;
}

static int is_proc_task_comm(const char* path) {
  if (!path) return 0;
  if (strncmp(path, "/proc/self/task/", 16) != 0) return 0;
  char* tail = strchr(path + 16, '/');
  return (tail != NULL && strcmp(tail, "/comm") == 0);
}

// open() - the original hook, still needed as fallback

int (*libc_open)(const char* path, int flags, ...) = NULL;

int open(const char* path, int flags, ...) {
  if (!libc_open) libc_open = dlsym(RTLD_NEXT, "open");

  mode_t mode = 0;
  va_list _args_;
  va_start(_args_, flags);
  if (flags & O_CREAT) mode = va_arg(_args_, int);
  va_end(_args_);

  if (is_nvidia_uvm(path) || is_proc_task_comm(path))
    return libc_open("/dev/null", flags, mode);
  return libc_open(path, flags, mode);
}

services:
  gpu-app:
    image: docker.io/library/rockylinux:9.3
    platform: linux/amd64
    container_name: rocky-gpu
    os: linux
    network_mode: host
    devices:
      - /dev/nvidia0:/dev/nvidia0
      - /dev/nvidiactl:/dev/nvidiactl
      - /dev/nvidia-modeset:/dev/nvidia-modeset
    volumes:
      # Individual library mounts to prevent glibc contamination
      - /compat/linux/usr/lib64/libnvidia-ml.so.1:/usr/lib64/libnvidia-ml.so.1:ro
      - /compat/linux/usr/lib64/libcuda.so.1:/usr/lib64/libcuda.so.1:ro
      - /compat/linux/usr/lib64/libnvidia-ptxjitcompiler.so.1:/usr/lib64/libnvidia-ptxjitcompiler.so.1:ro
      
      # Binary and Shim
      - /compat/linux/usr/bin/nvidia-smi:/usr/local/bin/nvidia-smi:ro
      - ./dummy-uvm.so:/usr/lib64/dummy-uvm.so:ro
      
      # Essential FreeBSD Linuxulator / Jail mounts
      - /compat/linux/proc:/proc 
      - /compat/linux/sys:/sys
      - /compat/linux/dev/fd:/dev/fd
      
      # SHM via named volume (Fixes ocijail /dev/shm bug)
      - rocky-shm:/dev/shm

    environment:
      - LD_LIBRARY_PATH=/usr/lib64
    
    entrypoint: ["/bin/sh", "-c"]
    command: 
      - |
        # Apply the jail shim
        echo '/usr/lib64/dummy-uvm.so' > /etc/ld.so.preload
        
        # Run the actual command
        /usr/local/bin/nvidia-smi

volumes:
  rocky-shm:
    driver: local
    driver_opts:
      type: tmpfs
      device: tmpfs
      o: "size=1g,mode=1777"

compose.yaml for ffmpeg nvenc encoding

#+begin_src yaml
services:
  gpu-app:
    image: docker.io/library/rockylinux:9.3
    platform: linux/amd64
    container_name: rocky-ffmpeg
    os: linux
    network_mode: host
    stdin_open: true
    tty: true
    volumes:
      # Mount the whole host lib directory to a NEUTRAL sub-folder
      - /compat/linux/usr/lib64:/usr/lib64/nvidia-host:ro

      # Binary, Shim, and User Scripts
      - /compat/linux/usr/bin/nvidia-smi:/usr/local/bin/nvidia-smi:ro
      - ./dummy-uvm.so:/usr/lib64/dummy-uvm.so:ro
      
      # Using /home/podman style paths to match the user
      - ./bin:/home/podman/bin:ro
      - ~/mnt:/home/podman/mnt:rw
      
      # Linuxulator / Jail essentials
      - /compat/linux/proc:/proc
      - /compat/linux/sys:/sys
      - /compat/linux/dev/fd:/dev/fd
      - rocky-shm:/dev/shm

    environment:
      # GLOBAL FIX: Set the path for everyone in the container
      - LD_LIBRARY_PATH=/usr/lib64
      - PATH=/usr/local/sbin:/usr/local/bin:/usr/sbin:/usr/bin:/sbin:/bin:/home/podman/bin
    
    entrypoint: ["/bin/sh", "-c"]
    command: 
      - |
        # 1. Setup GPU Shim
        echo '/usr/lib64/dummy-uvm.so' > /etc/ld.so.preload

        # We create 'video_host' with GID 44 to match FreeBSD's video group
        groupadd -g 44 video_host || true
        groupadd -g 1001 podman || true

        # create podman user with same id as on freebsd
        # add to root, wheel and the video group on podmad and freebsd
        useradd -u 1001 -g 1001 -G 0,10,39,44 -m podman || true

        # 3. Setup Shell Environment
        cp /etc/skel/.bashrc /home/podman/.bashrc
        cp /etc/skel/.bash_profile /home/podman/.bash_profile
        
        # Add the vars to .bashrc just to be safe for interactive sessions
        {
          echo 'export LD_LIBRARY_PATH=/usr/lib64:/usr/lib64/nvidia-host'
          echo 'export LD_PRELOAD=/usr/lib64/dummy-uvm.so'
          echo 'export PATH=$PATH:/home/podman/bin'
        } >> /home/podman/.bashrc

        # Ownership for internal home files only (no -R to protect host mounts)
        chown podman:podman /home/podman /home/podman/.bashrc /home/podman/.bash_profile

        # 4. Install FFmpeg
        dnf install -y https://mirrors.rpmfusion.org/free/el/rpmfusion-free-release-9.noarch.rpm \
