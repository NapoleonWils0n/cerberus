#+STARTUP: showall
* fdkaac 
** vbr bitrate

#+begin_src sh
fdkaac -m 5 input.wav 
#+end_src

#+begin_example
-w, --bandwidth <n>
       Frequency bandwidth (lowpass cut-off frequency) in Hz.  Available on AAC LC only.
#+end_example

#+begin_src sh
fdkaac -m 5 -w 18000 input.wav 
#+end_src

output

#+begin_src sh
fdkaac -m 5 input.wav -o output.m4a
#+end_src

** convert wav to m4a

#+begin_src sh
fdkaac -m 5 -w 18000 -o output.m4a input.wav
#+end_src

** pipe a file into fdkaac

#+begin_src sh
ffmpeg -i input.wav -vn -f caf - | fdkaac -m 5 -w 18000 -o output.aac -
#+end_src
