digraph {
    label="NotebookLM Workflow";
    labelloc="t";
    fontsize=20; // Adjust for visibility
    fontname="Helvetica-Bold";

    rankdir=LR;
    splines=ortho;
    nodesep=0.4; 
    ranksep=1.5; 
    
    node [shape=box, style=filled, fillcolor="white", fontname="Helvetica", fontsize=11];

    // 1. DATA SOURCES
    subgraph cluster_inputs {
        label = "1. Data Sources";
        style = "filled";
        fillcolor = "#fff9c4";
        youtube [label="YouTube URLs"];
        av_files [label="Audio/Video Files"];
        images [label="Images (OCR)"];
        web_search [label="Web Search"];
        urls [label="Website URLs"];
        g_drive [label="Google Drive"];
        copied_text [label="Copied Text"];
    }

    // 2. AI PROCESSING
    subgraph cluster_processing {
        label = "2. AI Processing";
        style = "filled";
        fillcolor = "#e1f5fe";
        transcript [label="Generated Transcript"];
        ocr_text [label="Extracted Text (OCR)"];
    }

    notebooklm [label="NotebookLM Core", shape=diamond, fillcolor="#bbdefb", style="filled,bold"];

    // 3. STUDIO TOOLS
    subgraph cluster_studio {
        label = "3. NotebookLM Studio";
        style = "filled";
        fillcolor = "#f3e5f5";
        
        subgraph cluster_creative {
            label = "Creative Suite (Custom Prompts)";
            style="dashed";
            color="#7e57c2";
            flashcards [label="Flashcards"];
            quiz [label="Quiz"];
            slides [label="Slide Deck"];
            data_table [label="Data Table"];
            audio_over [label="Audio Overview"];
            video_over [label="Video Overview"];
            infographics [label="Infographics"];
        }
        
        mind_map [label="Mind Map"];
        reports [label="Reports"];
        notes [label="Notes"];
    }

    // 4. EXPORT OPTIONS
    subgraph cluster_export {
        label = "4. Export Options";
        style = "filled";
        fillcolor = "#c8e6c9";
        g_docs [label="Google Docs"];
        pdf [label="PDF"];
        wav [label="Audio (.wav)"];
        mp4 [label="Video (.mp4)"]; // New Video Node
        link [label="Shared Link"];
    }

    // --- REFINED COMPACT LEGEND ---
    subgraph cluster_legend {
        label = "Flow Legend";
        fontsize = 10;
        legend_node [
            shape=plaintext, 
            label="Orange: Multimedia / OCR Path\lBlue: Standard Text Path\lGreen: Document Export Path\lPurple: Audio/Video Export Path\l"
        ];
    }

    // CONNECTIONS
    
    // Multimedia Processing (Orange)
    edge [color="#e67e22", penwidth=2];
    { youtube av_files } -> transcript;
    images -> ocr_text;
    
    // Ingestion to Core (Blue)
    edge [color="#3498db", penwidth=1.5];
    { web_search urls g_drive copied_text transcript ocr_text } -> notebooklm;

    // Core to Studio (Grey)
    edge [color="#7f8c8d", penwidth=1];
    notebooklm -> { audio_over video_over flashcards quiz slides data_table mind_map reports notes infographics };

    // Document Exports (Green)
    edge [color="#27ae60", penwidth=1.5];
    { reports notes flashcards quiz infographics } -> g_docs;
    { reports slides data_table infographics } -> pdf;
    
    // Media Exports (Purple)
    edge [color="#8e44ad", penwidth=1.5];
    audio_over -> wav;
    video_over -> mp4; // Correctly mapped to Video Export
    
    // Web Sharing (Light Blue)
    edge [color="#2980b9"];
    notes -> link;
}
