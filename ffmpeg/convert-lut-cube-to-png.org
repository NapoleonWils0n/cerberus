#+STARTUP: content
* convert lut cube to png
** convert a single lut cube file into a png

#+begin_src sh
ffmpeg -f lavfi -i haldclutsrc=8 -vf "lut3d='lut.cube'" -frames:v 1 lut.png
#+end_src

** use fd to batch convert .cube files to .png

#+begin_src sh
fd -e cube -x ffmpeg -f lavfi -i haldclutsrc=8 -vf "lut3d='{}'" -frames:v 1 "{.}.png"
#+end_src

** use a for loop to batch convert .cube files to .png

#+begin_src sh
for f in *.cube; do 
  ffmpeg -f lavfi -i haldclutsrc=8 -vf "lut3d='$f'" -frames:v 1 "${f%.cube}.png"; 
done
#+end_src
